\begin{equation}
  \begin{gathered}
    x_{i, l} = m_{i}(u_{i, l}) + \epsilon_{i, l}, \\
    m_{i}(u_{i, l}) =
      \eta_{0, i} +
      \eta^{b}_{1, i}(u_{i, l} - \kappa_{i}) \mathbbm{1}_{\{u_{i, l} < \kappa_{i}\}} +
      \eta^{a}_{1, i}(u_{i, l} - \kappa_{i}) \mathbbm{1}_{\{u_{i, l} \geq \kappa_{i}\}}.
  \end{gathered}
  \label{eqn:piecewise-fluid-model}
\end{equation}
